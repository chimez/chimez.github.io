用Rust写Web

编译到wasm的工具

rustc

使用基本编译器进行编译,目前编译器版本要>1.25.0,所以要先升级到开发版rustup install nightly;rustup default nightly

cargo-web

使用cargo管理项目,先安装插件cargo-webx,cargo install -f cargo-web 配置样例: 

default-target = "wasm32-unknown-unknown"
prepend-js = "src/runtime.js"

[cargo-web]
minimum-version = "0.6.0" #cargo-web version

[target.emscripten]
prepend-js = "src/emscripten_runtime.js"
# This will enable Emscripten's SDL2 port. Consult Emscripten's documentation
# for more details.
link-args = ["-s", "USE_SDL=2"]

[target.wasm32-unknown-unknown]
prepend-js = "src/native_runtime.js"
