       (define (odd? n)
         (not (even? n)))
       (define (even? n)
         (if (= n 0)
             #t
             (odd? (- n 1))))

1.2 栈方法

       (define ss '())
       (define (push x)
         (set! ss (cons x ss)))
       (define (pop)
         (let* ((x (car ss)))
           (set! ss (cdr ss))
           x))

       (define (stack-odd? n)
         (let loop ((i n))
           (if (not (= i 0))
               (begin
                 (push i)
                 (loop (- i 1)))
               (push #f)))
         (let ((flag #t))
           (while (not (null? ss))
             (pop)
             (set! flag (not flag)))
           flag))
       (define (stack-even? n)
         (not (stack-odd? n)))

       (stack-odd? 3010)

#f

2.2 尾递归版

尾递归就要将中间计算过程输入到参数中去,代价就是要给个初值 这里要求初值必须是 true 

       (declare user-even?) 
       (defn user-odd? [[n flag]]
         (user-even? [n (not flag)]))
       (defn user-even? [[n flag]]
         (if (<= n 0)
           (if flag
             true
             false)
           (user-odd? [(- n 1) flag])))

       (println (user-odd? [31 true]))
       (println (user-even? [31 true]))
       (println (user-odd? [32 true]))
       (println (user-even? [32 true]))

2.3 trampoline 优化多重尾递归

trampoline 只要函数返回的还是函数, 就会一直调用它,直到返回非函数为止 给的参数只传递给调用的函数,不向次级函数传递值

2.3.1 上面函数的例子

        (declare user-even?)
        (defn user-odd? [[n flag]]
          #(user-even? [n (not flag)]))
        (defn user-even? [[n flag]]
          (if (<= n 0)
            (if flag
              true
              false)
            #(user-odd? [(- n 1) flag])))

        (println (trampoline user-odd? [31 true]))
        (println (trampoline user-odd? [32 true]))
        (println (trampoline user-even? [31 true]))
        (println (trampoline user-even? [32 true]))

2.3.2 有限状态自动机

        (defn foo [cmds]
          (letfn
              [(a-> [[_ & rs]]
                 #(case _ 
                    :a-b (b-> rs)
                    :a-c (c-> rs)
                    false))
               (b-> [[_ & rs]]
                 #(case _ 
                    :b-a (a-> rs)
                    :b-c (c-> rs)
                    false))
               (c-> [[_ & rs]]
                 #(case _ 
                    :c-a (a-> rs)
                    :c-b (c-> rs)
                    :final (do (println "!") true)
                    false))]
            (trampoline a-> cmds)))

2.4 栈方法

       (def ss (atom []))
       (defn user-push [x]
         (swap! ss #(cons x %)))
       (defn user-pop []
         (let [x (first @ss)]
           (swap! ss #(rest %))
           x))

       (defn stack-odd? [n]
         (loop [i n]
           (if (not (= i 0))
             (do
               (user-push i)
               (recur (- i 1)))
             (user-push false)))
         (let [flag (atom true)]
           (while (not (empty? @ss))
             (user-pop)
             (swap! flag #(not %)))
           @flag))
       (defn stack-even? [n]
         (not (stack-odd? n)))

       (println (stack-odd? 31))
       (println (stack-odd? 32))
       (println (stack-even? 31))
       (println (stack-even? 32))
