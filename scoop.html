4.1 psfzf

只安装fzf不够，还要再安装PSFzf 才能用快捷键。 参考 Make Your PowerShell 7 Truly Powerful 

Install-Module PSFzf -Scope CurrentUser
Remove-PSReadlineKeyHandler 'Ctrl+r'
Remove-PSReadlineKeyHandler 'Ctrl+t'
Import-Module PSFzf

5 配置文件

# 代理
function proxy_start {
    param (
    )
    Set-Variable -Name "http_proxy" -Value "http://127.0.0.1:8081" -Scope "Global" -Visibility "Public"
    Set-Variable -Name "https_proxy" -Value "http://127.0.0.1:8081" -Scope "Global" -Visibility "Public"
}

function proxy_stop {
    param (
    )
    Set-Variable -Name "http_proxy" -Value "" -Scope "Global" -Visibility "Public"
    Set-Variable -Name "https_proxy" -Value "" -Scope "Global" -Visibility "Public"
}

# 设置 tab 补全
Set-PSReadlineKeyHandler -Key Tab -Function Complete

# 启用 starship 美化
Invoke-Expression (&starship init powershell)

# 命令别名
Set-Alias l lsd

# 配置 psfzf
Set-PsFzfOption -PSReadlineChordProvider 'Ctrl+t' -PSReadlineChordReverseHistory 'Ctrl+r'
