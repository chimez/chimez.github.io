vimwiki

vimwiki是vim的一个插件,可以用来生成个人的wiki,日记和todolist

安装与基本配置

安装

" vimwiki
Plug 'vimwiki/vimwiki', { 'branch': 'dev' }

" 日历,用于日记
Plug 'mattn/calendar-vim'

基本配置

" vimwiki 
let g:vimwiki_list = [{'path': '~/Documents/vimwiki', 
										\ 'path_html': '~/Documents/vimwiki/public_html',
										\ 'index': 'index',
										\ 'template_path': '~/Documents/vimwiki/public_html/templates/',
										\ 'template_default': 'def_template',
										\ 'template_ext': '.html',
										\ 'css_name': '~/Documents/vimwiki/public_html/css/main.css',
										\ 'auto_export': 1,
										\ 'auto_toc': 1}]
										
" 对中文用户来说，我们并不怎么需要驼峰英文成为维基词条
let g:vimwiki_camel_case = 0
" 是否在计算字串长度时用特别考虑中文字符
let g:vimwiki_CJK_length = 1
" 设置对于目录格式的行为
let g:vimwiki_dir_link = 'index'
" 自定义html,防止被删掉    
let g:vimwiki_user_htmls = '404.html,search.html'

基本使用

基本指令

指令 功能
<leader>ww 开启笔记主页
<leader>wi 开启日记主页
<leader>w<leader>i 更新日记列表(只能在日记主页中使用)
<leader>w<leader>w 新建今天的日记
<leader>whh 生成网页并在浏览器中打开
:Calendar 打开日历
:VimWikiTOC 自动创建目录
:VimWikiAll2Html 全部生成网页
在路径上 回车键 新建路径并进入子页面
在字页面 退格键 返回上个页面

wiki语法

语法 含义
%% 注释
%title Title 网页标题
=...= 标题
{{{}}} 代码
