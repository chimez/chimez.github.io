插值拟合

yi=interp1(x,Y,kind) 其中，yi为根据插值算法求得的函数。 x和Y为已知的数据点向量。参量用于确定具体的插值方法，包括：

kind 表示
linear 采用线性插值方法
cubic 采用三次插值方法
nearest 采用最近点插值方法 
# 导入基本库
import numpy as np
from scipy import interpolate
import matplotlib.pyplot as plt

# 创建数据

nx = np.linspace(-np.pi, np.pi, 30)  # 生成数据点(起点,终点, 个数)
ny = f(nx)

#绘图
plt.plot(x,y,nx, ny)
plt.show()

# 导入基本库

from scipy import interpolate
import matplotlib.pyplot as plt

    from pylab import mpl
    mpl.rcParams['font.sans-serif'] = [font_name]
    mpl.rcParams['axes.unicode_minus'] = False 
conf_zh('SimHei')


def plot_it(x,y,title,x_label,y_label,xmin,xmax):
    # 插值拟合
    f = interpolate.interp1d(x, y, kind="cubic") # 创建一个拟合函数(原x,原y,kind=模式)
    # 多项式拟合
    #z = np.polyfit(np.array(x), np.array(y), 5)
    #f = np.poly1d(z)
    # 生成拟合曲线点
    nx = np.linspace(xmin,xmax,500)
    ny = f(nx)
    
    # 拟合后曲线区间分析
    # 寻找左右点位置
    half=0.8*ny.max() # y值目标
    for i in ny:
        if i>half:
            xnum_small=ny.tolist().index(i) # 左侧值序号
            break
    for i in range(0,len(ny.tolist()))[::-1]:
        if ny.tolist()[i]>half:
            xnum_large=i # 右侧值序号
            break
    x_highest=nx.tolist()[ny.tolist().index(ny.max())] # 最大值x坐标
    x_small=nx.tolist()[xnum_small] # 左侧值x坐标
    plt.axvline(x_highest,color="black") #  画竖直线
    plt.axvline(x_small,color="black")
    plt.plot([0,10],[half,half],'k:') # 画水平线
    plt.text(x_small+(x_highest-x_small)/2,half*0.9,str(round(x_highest-x_small,4)),fontsize=16)  #标注区间大小
    
    # 设置图片格式
    #plt.figure(figsize=(8,4)) # 设置图片尺寸 800px X 400px
    plt.xlabel(x_label) # x轴标签
    plt.ylabel(y_label) # y轴标签
    plt.title(title) # 图片标签
    plt.xlim(xmin,xmax)
    
    # 画线
    plt.plot(x,y,'ko',label='点') # 数据点
    plt.plot(nx,ny,'k-',label='拟合线') # 拟合线
    plt.text(0,0,'0')
    # 显示图像
    plt.legend(bbox_to_anchor=(1,1)) #显示标注并设置标注位置
    #plt.savefig('/home/bacon/'+title+'.png') # 保存图像到文件, 一定要先保存后显示,否则会出错
    plt.show() # 显示图片

绘制坐标轴图

# 导入基本库
import numpy as np
from scipy import interpolate
import matplotlib.pyplot as plt
import sympy as sp
from mpl_toolkits.axes_grid.axislines import SubplotZero
# 中文显示
def conf_zh(font_name):
    from pylab import mpl
    mpl.rcParams['font.sans-serif'] = [font_name]
    mpl.rcParams['axes.unicode_minus'] = False 
conf_zh('SimHei')


def plot_it(x,y,title,x_label,y_label,xmin=0,xmax=1,ymin=0,ymax=1):
    # 设置图片格式
    fig = plt.figure(1)
    ax = SubplotZero(fig, 111)
    fig.add_subplot(ax)
    for direction in ["xzero", "yzero"]:
        ax.axis[direction].set_axisline_style("-|>")
        ax.axis[direction].set_visible(True)
    for direction in ["left", "right", "bottom", "top"]:
        ax.axis[direction].set_visible(False)
    
    #plt.title(title) # 图片标签
    plt.ylim(ymin,ymax)
    plt.xticks([])
    plt.yticks([])
    # 画线
    ax.plot(x,y) # 数据点
    z=[]
    for i in y:
        z.append(-i)
    ax.plot(x,z) # 数据点
    plt.text(0,0,'0',fontsize=16)
    plt.text(xmax,0.1,x_label,fontsize=16) #x
    plt.text(-0.5,ymax+0.1,y_label,fontsize=16) #y
    plt.text(-0.3,1.2,"$w_{0}$",fontsize=16) #字
    # 显示图像
   # plt.savefig('/home/bacon/'+title+'.png') # 保存图像到文件, 一定要先保存后显示,否则会出错
    plt.show() # 显示图片

def w(z):
    return (1+(z)**2)**(1/2)
x=list(range(0,4000))
y=[]
for i in range(0,len(x)):
    x[i]=x[i]/1000-2
    y.append(w(x[i]))
plot_it(x,y,"高斯光束与参数","$z$","$w(z)$",0,x[-1],-3,3)

plot 线型设置

plot画图时可以设定线条参数。包括：颜色、线型、标记风格。

控制颜色

颜色之间的对应关系为

标记 b c g k m r w y
颜色 蓝色 青色 绿色 黑色 紫红色 红色 白色 黄色

有三种表示颜色的方式:

  1. 用全名
  2. 16进制如：#FF00FF
  3. RGB或RGBA元组（1,0,1,1）
  4. 灰度强度如：‘0.7’

控制线型

标记 - -- -. :
线型 实线 短线 点画线 虚线

Legend中指定线和图例相对应的方法

在plot时的返回参数是两个,所以加逗号忽略第二个参数,用第一个来加图例才行 

a, = plt.plot(x,y)
plt.legend([a],["a"])

并行化画图时,解决爆内存和同一图中画了好多线的方法

  1. 利用随机数做画图编号
  2. 画完图要关闭这个图 
from random import randint
fig_num = randint(0,10**10)
fig = plt.figure(fig_num)
cop, = plt.plot(tlist,na,label="$a^\\dagger a$")
z2p, = plt.plot(tlist,co,label="concurrence")
plt.legend([cop,z2p],["$a^\\dagger a$","concurrence"])
fig = plt.gcf()
fig.savefig('filename.png')
plt.close(fig_num)

让画出来的图片连在一起动起来的方法

matplotlib本身并不能实现这个功能,可以采用的方法有ffmpeg, opencv, imagemagic等,这里选用最简单方便的imagemagic 

convert -delay 10 -loop 0 a1.png a2.png a.gif

选项含义:

选项 意义
-delay 10 每张图片延时
-loop 0 循环播放次数,0为一直循环
