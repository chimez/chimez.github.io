1 Nim 学习笔记

1.1 内存管理

1.1.1 手动分配内存

  1. 使用 alloc0 创建内存
  2. 创建的内存是无类型的, 要用 cast 转换成需要的类型
  3. 用 dealloc 释放内存
      const size:int = 1600*1600
      var A = cast[ptr array[0..size-1, float64]](alloc0(sizeof(float64)*size))
      A[1] = 1.2
      echo A[]
      dealloc(A)

1.1.2 使用 GC 自动管理内存

  1. 使用 ref object 创建自动管理的对象
      const size:int = 1600*1600
      type ta = ref object
        data:array[0..size-1, float64]

      var A:ta
      new(A)

      for i in 0..size-1:
        A.data[i] = (float64)i
        echo len(A.data)

1.2 手动编译

参考地址： [https://nim-lang.org/docs/backends.html]

  1. 生成代码

    1. 整个应用: 需要指定代码存储位置 nimcache, 以及头文件 header 

                 nim cpp --path:./src --passc:"-I/usr/include/eigen3" --nimcache:cc1 --header tests/test_matrix.nim

    2. 链接库： 需要指定代码存储位置，无 main 头文件 

                 # 动态链接
           nim c --noMain --noLinking --header:fib.h fib.nim
           # 静态链接
           nim c --app:staticLib --noMain --header fib.nim

  2. 编译： 需要指定代码存储位置，nim库，和编译用的其它东西 

            # 单纯代码
        g++  -Ipath/to/nimcache -Ipath/to/nim/lib $HOME/.cache/nim/fib_d/*.c maths.c
        # 静态链接
        gcc -o m -Inimcache -Ipath/to/nim/lib libfib.nim.a maths.c

2 FFI

2.1 从 nim 调用 c 函数

  1. 引入库
    1. 头文件 {.passL: "-lgsl"}
    2. 动态链接
    3. 静态链接
  2. 函数 proc f(){.header:"h", importc: "f", varargs.}
    1. 有参数
    2. 无参数
  3. 结构体
  4. 枚举
  5. 指针
     # include lib header
     {.passL: gorge("pkg-config --libs gsl").}

     # functions
     # function with multi args
     proc printf(formatstr: cstring) {.header: "<stdio.h>", importc: "printf", varargs.}
     # function with return type
     proc gsl_sf_bessel_J0(x: cdouble): cdouble {.header: "<gsl/gsl_sf_bessel.h>", importc: "$1"}

     # use
     const x = 5.0
     let  y:cdouble = gsl_sf_bessel_J0(x)
     echo y

     # use without returns
     printf("J0(%g) = %.18e\n", x, y)

2.2 c2nim: 将c的头文件转换成 nim 文件

  1. 预处理器似乎不好使,所以可以先 gcc a.h -E -P -C -o a_new.h 展开宏,再运行转换 c2nim a_new.h
  2. 要先注释掉标准库的 #include <stdlib.h>, #include <stdio.h> 等

2.3 从 nim 调用 c++

2.3.1 方法

  1. 模板类 object
  2. 构造函数: constructor 不可用, 只能使用 new 方法
  3. 运算符重载方法: 包裹成指针来使用

2.3.2 注意

  1. 只能到处传指针, 别的方法都不好使
  2. 不能访问私有变量, 只能使用公有的
  3. 数字泛型 static[int] 在类型声明里好事, 函数声明里不好使, 只能手动实例化

2.3.3 普通模板类

      {.emit"""/*TYPESECTION*/
      template <typename T>
      class Complex{

      public:
          //构造函数
          Complex(T a, T b)
          {
              this->a = a;
              this->b = b;
          }

          //运算符重载
          Complex<T> operator+(Complex &c)
          {
              Complex<T> tmp(this->a+c.a, this->b+c.b);
              return tmp;
          }

      T a;
          T b;
      };

      """.}

      type Complex[T] {.importcpp:"Complex".} = object
          a,b:T

      proc newComplex[T](a,b:T):ptr Complex[T] {.importcpp:"new Complex<'**0>(@)", nodecl.}
      proc `+`[T](a,b:Complex[T]):ptr Complex[T]{.importcpp:"&(# + #)"}
      let c= newComplex[int](1,2)

      echo c[].a
      echo c[].b
      echo (c[] + c[]).a

2.3.4 数字参数模板类的构造函数

      {.emit: """/*TYPESECTION*/
      template <int N, typename T>
      class AAAA {
      public:
        T data[N];
        AAAA(int a){
          this->data[0] = a;
      }
      };

      """ .}

      type AAAAType {.importcpp: "AAAA<'0, '1>".} [N: static[int]; T] = object
          data: array[N, T]

      proc cc[T](a:int):ptr AAAAType[1, T]{.importcpp:"new AAAA<1, '1>(@)", nodecl.}

      let aa = cc[int](1)
